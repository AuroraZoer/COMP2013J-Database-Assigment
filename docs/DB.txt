--run these commands under the root account:
--Create a user "dnb" with password "password"
--this will be used by the JDBCTool class to access the database.
CREATE USER 'dnb'@'localhost' IDENTIFIED WITH mysql_native_password BY 'password';
--create the commodity management platform database:
CREATE DATABASE cmp;
--grant editing privilege on the CMP database to our account:
GRANT ALL ON cmp.* TO 'dnb'@'localhost' WITH GRANT OPTION;

--the following can be executed by the new account:
CREATE TABLE user(
    uid INT,
    name VARCHAR(100),
    account INT,
    PRIMARY KEY(uid)
    );
CREATE TABLE account(
    aid INT,
    login VARCHAR(100),
    password VARCHAR(100),
    PRIMARY KEY(aid)
    );
-- representing the commodity stored in warehouses.
CREATE TABLE commodity(
    cid INT,
    itemName VARCHAR(100),
    type VARCHAR(100),
    price FLOAT,
    stock INT,
    PRIMARY KEY(cid)
    );
CREATE TABLE sale(
    id INT,
    time DATETIME,
    itemName VARCHAR(100),
    quantity INT,
    soldFor FLOAT,
    PRIMARY KEY(id)
    );
